REM --- Example 1: Basic function, no arguments ---
FUNCTION greeting()
STRING Hello, World!
END_FUNCTION

REM --- Example 3: Function with a parameter, then called ---
FUNCTION greetings_user(VAR1)

END_FUNCTION

greeting()
DELAY 1000
ENTER
greetings_user("USER987654321")
ENTER
